	desc = "EVENTDESC602001"
	fire_only_once = no

	trigger {
		NOT = { has_province_modifier = major_port }
		owner = { freedom_of_trade = 1 }
		has_building = trading_port
	}

	mean_time_to_happen = {
		months = 36
	}

	option = {
		name = "EVTOPTA602001"
		state_scope = {
			any_owned = {
    			limit = {
        			has_building = trading_port
    			}
    			add_province_modifier = { name = major_port duration = -1 }
			}
		}
		ai_chance = { factor = 100 }
	}
}
